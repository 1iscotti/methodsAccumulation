const a = '[河北省:(廊坊市:安次区;广阳区;)(唐山市:)(保定市:)] ';
1. 匹配xx市
const reg1 = /[^\(]+.市(?=:)/g
a.match(reg1)
// ["廊坊市", "唐山市", "保定市"]
2. 匹配xx省
const reg2 = /[^\[]+.省(?=:)/g
a.match(reg2)
// ["河北省"]
